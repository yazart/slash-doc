#!/usr/bin/env node

'use strict';

const fs = require('node:fs/promises');
const path = require('node:path');

const ARCHIVE_HEADER = '=== NODE20 DIRECTORY ARCHIVE v1 ===\n';
const FILE_PREFIX = '=== FILE ';
const FILE_SUFFIX = ' ===\n';
const FILE_END = '\n=== END FILE ===\n';

const IMAGE_EXTENSIONS = new Set([
  '.avif',
  '.bmp',
  '.gif',
  '.heic',
  '.heif',
  '.ico',
  '.jpeg',
  '.jpg',
  '.png',
  '.svg',
  '.tif',
  '.tiff',
  '.webp',
]);

function printUsage() {
  console.log(`
Использование:
  node concat-files.js concat <директория> <архив>
  node concat-files.js extract <архив> <директория>

Примеры:
  node concat-files.js concat ./project ./project.bundle.txt
  node concat-files.js extract ./project.bundle.txt ./restored-project
`.trim());
}

async function collectFiles(rootDirectory, excludedFile) {
  const files = [];

  async function walk(currentDirectory) {
    const entries = await fs.readdir(currentDirectory, { withFileTypes: true });
    entries.sort((left, right) => left.name.localeCompare(right.name));

    for (const entry of entries) {
      const absolutePath = path.join(currentDirectory, entry.name);

      if (entry.isDirectory()) {
        await walk(absolutePath);
      } else if (entry.isFile() && absolutePath !== excludedFile) {
        files.push(absolutePath);
      }
    }
  }

  await walk(rootDirectory);
  return files;
}

function isUtf8(buffer) {
  return Buffer.from(buffer.toString('utf8'), 'utf8').equals(buffer);
}

function toPortableRelativePath(rootDirectory, filePath) {
  return path.relative(rootDirectory, filePath).split(path.sep).join('/');
}

async function concatenate(inputDirectory, archivePath) {
  const rootDirectory = path.resolve(inputDirectory);
  const outputFile = path.resolve(archivePath);
  const rootStats = await fs.stat(rootDirectory);

  if (!rootStats.isDirectory()) {
    throw new Error(`Не является директорией: ${rootDirectory}`);
  }

  const files = await collectFiles(rootDirectory, outputFile);
  const chunks = [Buffer.from(ARCHIVE_HEADER, 'utf8')];

  for (const filePath of files) {
    const source = await fs.readFile(filePath);
    const extension = path.extname(filePath).toLowerCase();
    const encoding = IMAGE_EXTENSIONS.has(extension) || !isUtf8(source)
      ? 'base64'
      : 'utf8';
    const payload = encoding === 'base64'
      ? Buffer.from(source.toString('base64'), 'ascii')
      : source;
    const metadata = {
      path: toPortableRelativePath(rootDirectory, filePath),
      encoding,
      length: payload.length,
    };

    chunks.push(
      Buffer.from(`${FILE_PREFIX}${JSON.stringify(metadata)}${FILE_SUFFIX}`, 'utf8'),
      payload,
      Buffer.from(FILE_END, 'utf8'),
    );
  }

  await fs.mkdir(path.dirname(outputFile), { recursive: true });
  await fs.writeFile(outputFile, Buffer.concat(chunks));
  console.log(`Готово: ${files.length} файл(ов) записано в ${outputFile}`);
}

function readLine(buffer, offset) {
  const lineEnd = buffer.indexOf(0x0a, offset);

  if (lineEnd === -1) {
    throw new Error('Поврежденный архив: не найден конец строки');
  }

  return {
    line: buffer.subarray(offset, lineEnd + 1).toString('utf8'),
    nextOffset: lineEnd + 1,
  };
}

function resolveSafeOutputPath(outputDirectory, portablePath) {
  if (
    typeof portablePath !== 'string'
    || portablePath.length === 0
    || portablePath.includes('\0')
    || path.posix.isAbsolute(portablePath)
  ) {
    throw new Error(`Недопустимый путь в архиве: ${String(portablePath)}`);
  }

  const root = path.resolve(outputDirectory);
  const destination = path.resolve(root, ...portablePath.split('/'));
  const relative = path.relative(root, destination);

  if (relative.startsWith('..') || path.isAbsolute(relative)) {
    throw new Error(`Путь выходит за директорию распаковки: ${portablePath}`);
  }

  return destination;
}

async function extract(archivePath, outputDirectory) {
  const archive = await fs.readFile(path.resolve(archivePath));
  const expectedHeader = Buffer.from(ARCHIVE_HEADER, 'utf8');

  if (!archive.subarray(0, expectedHeader.length).equals(expectedHeader)) {
    throw new Error('Неизвестный формат архива или поврежденный заголовок');
  }

  let offset = expectedHeader.length;
  let extractedCount = 0;
  const expectedEnd = Buffer.from(FILE_END, 'utf8');

  while (offset < archive.length) {
    const { line, nextOffset } = readLine(archive, offset);

    if (!line.startsWith(FILE_PREFIX) || !line.endsWith(FILE_SUFFIX)) {
      throw new Error(`Поврежденный разделитель файла около байта ${offset}`);
    }

    const metadataText = line.slice(FILE_PREFIX.length, -FILE_SUFFIX.length);
    let metadata;

    try {
      metadata = JSON.parse(metadataText);
    } catch {
      throw new Error(`Поврежденные метаданные файла около байта ${offset}`);
    }

    if (
      !Number.isSafeInteger(metadata.length)
      || metadata.length < 0
      || !['utf8', 'base64'].includes(metadata.encoding)
    ) {
      throw new Error(`Недопустимые метаданные для файла ${metadata.path}`);
    }

    const payloadEnd = nextOffset + metadata.length;
    const endMarkerEnd = payloadEnd + expectedEnd.length;

    if (
      payloadEnd > archive.length
      || !archive.subarray(payloadEnd, endMarkerEnd).equals(expectedEnd)
    ) {
      throw new Error(`Повреждено содержимое файла ${metadata.path}`);
    }

    const payload = archive.subarray(nextOffset, payloadEnd);
    const contents = metadata.encoding === 'base64'
      ? Buffer.from(payload.toString('ascii'), 'base64')
      : payload;
    const destination = resolveSafeOutputPath(outputDirectory, metadata.path);

    await fs.mkdir(path.dirname(destination), { recursive: true });
    await fs.writeFile(destination, contents);

    extractedCount += 1;
    offset = endMarkerEnd;
  }

  console.log(
    `Готово: ${extractedCount} файл(ов) восстановлено в ${path.resolve(outputDirectory)}`,
  );
}

async function main() {
  const [command, source, destination] = process.argv.slice(2);

  if (!command || !source || !destination || !['concat', 'extract'].includes(command)) {
    printUsage();
    process.exitCode = 1;
    return;
  }

  if (command === 'concat') {
    await concatenate(source, destination);
  } else {
    await extract(source, destination);
  }
}

main().catch((error) => {
  console.error(`Ошибка: ${error.message}`);
  process.exitCode = 1;
});
